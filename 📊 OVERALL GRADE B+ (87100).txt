  BUG #1: Array Bounds Vulnerability

  File: GameManager_STEP_FNs.gml:589Severity: High

  function getChallengeData() {
      return challenge_data.CHALLENGES[global.Game.Challenge.current - 1];
      // ⚠️ No bounds checking!
  }

  Risk: Crash if global.Game.Challenge.current is 0 or > 8

  Fix:
  function getChallengeData() {
      var index = global.Game.Challenge.current - 1;
      if (index < 0 || index >= array_length(challenge_data.CHALLENGES)) {
          log_error("Invalid challenge index: " + string(global.Game.Challenge.current),
                    "getChallengeData", 3);
          return challenge_data.CHALLENGES[0]; // Fallback
      }
      return challenge_data.CHALLENGES[index];
  }
